// Heretic Ambience ---------------------------

:HereticAmbience
	// Heretic waits 10 seconds after level load before it starts
	// playing any sounds.
	delayonce	350

	volumerand	0  50
	attenuation	none
	randomsequence
	delayrand	210  465
	restart
end

// to make them stand-alone so you don't always have to play them
// from inside HereticAmbience.
