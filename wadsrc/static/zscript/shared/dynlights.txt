class DynamicLight : Actor native
{
	Default
	{
		Height 0;
		Radius 0.1;
		FloatBobPhase 0;
		+NOBLOCKMAP
		+NOGRAVITY
		+FIXMAPTHINGPOS
		+INVISIBLE
	}
}
